fake = read_csv("Fake.csv") %>% mutate(real="fake")
real = read_csv("True.csv") %>% mutate(real="real")

news = bind_rows(fake,real)

cleaned = news %>%
  unnest_tokens(word, text, token="words") %>%
  anti_join(stop_words) %>%
  filter(!str_detect(word, "[.:0-9]"))
